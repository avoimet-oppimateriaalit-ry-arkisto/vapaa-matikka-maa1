\begin{tehtava}
Ratkaise: \\
a) $ x^2 = 4 $ \qquad
b) $ x^3 = 27 $ \qquad
c) $ x^5 = -1 $ \qquad
d) $ x^2 - 3 = 0 $ \qquad
e) $ x^3 + 125 = 0 $
\begin{vastaus}
a) $ x = 2 $ \qquad
b) $ x = 3 $ \qquad
c) $ x = -1 $ \qquad
d) $ x = \sqrt{3} $ \qquad
e) $ x = -5 $ 
\end{vastaus}
\end{tehtava}

\begin{tehtava}
Ratkaise \\
a) $ 5x^2 = 25 $ \qquad
b) $ (2x)^3 = 8 $ \qquad
c) $ x^4 = \frac{1}{4} $ \qquad
d) $ (3x)^2 = 36 $ \qquad
e) $ (4x)^2 + 16 = 0 $.
\begin{vastaus}
a) $ x = \sqrt{5} $ \qquad
b) $ x = 1 $ \qquad
c) $ x = \pm\frac{1}{\sqrt{2}} $ \qquad
d) $ x = 2 $ \qquad
e) Ei ratkaisua. 
\end{vastaus}
\end{tehtava}

\end{tehtava}

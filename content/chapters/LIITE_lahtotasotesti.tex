Oheinen lähtötasotesti auttaa opettajaa selvittämään, mihin asioihin kurssilla kannattaa erityisesti perehtyä. Kirjoita koko ratkaisu välivaiheineen ja palauta testi opettajalle ensimmäisellä tunnilla arviointia varten. Testien tehtävien tekemiseen pitäisi kulua aikaa alle varttitunti.

\begin{tehtava}
	Sievennä. 
	\begin{alakohdat}
        \alakohta{$-2^4-(-2)^3$}
		\alakohta{$4\cdot \frac{2}{5} + \frac{2}{3}\cdot \frac{3}{5}$}
		\alakohta{$\frac{2}{5} : \frac{3}{2}$}
		\alakohta{$-2(4x^2-x)-x$}
		\alakohta{$\frac{ab^2}{a^3b}$}
	\end{alakohdat}
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{$-16-(-8)=-16+8=-8$}
			\alakohta{$\frac{8}{5} + \frac{2}{5}=\frac{10}{5} = 2$}
			\alakohta{$\frac{2}{5} \cdot \frac{2}{3}=\frac{4}{15}$}
			\alakohta{$-8x^2+x$}
			\alakohta{$\frac{b}{a^2}$}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Ratkaise yhtälöt.
	\begin{alakohdat}
		\alakohta{$2x+5 = -1$}
		\alakohta{$x^2 = 9$}
	\end{alakohdat}
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{$x=-3$}
			\alakohta{$x=-3$ tai $x=3$}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
Ratkaise epäyhtälö $2x\leq5x-1$.

	\begin{vastaus}
		$x\geq \frac{1}{3}$
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Funktiot $f$ ja $g$ ovat määritelty seuraavasti: $f(x)= x^2+3x$ ja $g(x)=2x-8$.
	\begin{alakohdat}
		\alakohta{Laske $f(-2)$.}
		\alakohta{Määritä funktion $g$ nollakohta.}
	\end{alakohdat}	
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{$(-2)^2+3\cdot(-2)=-2$}
			\alakohta{$x=4$}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Matka ja aika ovat suoraan verrannollisia. Aika ja nopeus on kääntäen verrannollisia. 
	\begin{alakohdat}
		\alakohta{Jos matka kaksinkertaistuu, niin mitä käy ajalle?}
		\alakohta{Jos nopeus kaksinkertaistuu, niin mitä käy ajalle?}
	\end{alakohdat}
	\begin{vastaus}
		\begin{alakohdat}
		\alakohta{Aika kaksinkertaistuu.}
		\alakohta{Aika puolittuu.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
Kaupassa on 15 prosentin alennusmyynti.
	\begin{alakohdat}
		\alakohta{Kuinka paljon 25,95 euron paita maksaa alennuksessa? Ilmoita vastaus pyörisettynä sentin tarkkuuteen.}
		\alakohta{Tuotteen alennettu hinta on 21,25 euroa. Mikä oli tuotteen alkuperäinen hinta?}
	\end{alakohdat}
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{22,06 euroa}
			\alakohta{25,00 euroa}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

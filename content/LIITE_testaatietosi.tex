\begin{tehtava}

	\begin{alakohdat}
		\alakohta{$(b-3a)(c+3)$}
		\alakohta{$(b-2a)(c+3)$}
		\alakohta{$2(b-a)(c+3)$}
		\alakohta{$(b-a)(2c+3$}
	\end{alakohdat}
    \begin{vastaus}

    \end{vastaus}
